CREATE TABLE User (    Email TEXT PRIMARY KEY,    First TEXT NOT NULL,    Last TEXT NOT NULL,    Password TEXT NOT NULL);CREATE TABLE Post (    PostID INTEGER PRIMARY KEY,    Title TEXT NOT NULL,    Content TEXT NOT NULL,    Likes INTEGER NOT NULL,    Email TEXT NOT NULL,    Label TEXT,    FOREIGN KEY (Email) REFERENCES User (Email),    FOREIGN KEY (Label) REFERENCES Community (Label));CREATE TABLE Comment (    CommentID INTEGER PRIMARY KEY,    PostID INTEGER NOT NULL,    Comment TEXT NOT NULL,    Email TEXT NOT NULL,    FOREIGN KEY (PostID) REFERENCES Post (PostID),    FOREIGN KEY (Email) REFERENCES User (Email));CREATE TABLE Community (    Label TEXT PRIMARY KEY,    Picture BLOB NOT NULL);
